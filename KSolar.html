KSolare 5GPro Connection
Model: KSY 3.4kW 1Ph 1MPPT (5G Pro)
Protocol: SolarmanV5 over TCP port 8899
COM1: WiFi datalogger stick
COM2: RS485 smart meter / wired backup
Modbus: 9600 baud · 8N1 · Slave ID 1
Setup Steps
1
Read the 10-digit serial number on the WiFi stick in COM1 — e.g. 2112345678
2
Connect stick to home WiFi via SolarmanPV app, or hotspot AP_XXXXXXXX → open 10.10.100.254
3
Find the inverter's LAN IP in your router device list. Look for "Hi-Flying" or open port 8899.
4
Verify: nmap -p 8899 <ip> — must show open.
5
Run Python poller (Code tab). Non-zero register values = connected successfully.

inverter_poller.py

#!/usr/bin/env python3 # KSolare 5GPro · SolarmanV5 Poller · Raspberry Pi 4 # pip install pysolarmanv5 paho-mqtt import time, json, logging from datetime import datetime from pysolarmanv5 import PySolarmanV5, NoSocketAvailableError import paho.mqtt.client as mqtt INVERTER_IP = "192.168.1.XXX" # from router device list LOGGER_SERIAL = 2112345678 # 10-digit from WiFi stick label RATED_W, TARIFF, POLL_S = 3400, 7.5, 30 mq = mqtt.Client(); mq.connect("localhost", 1883); mq.loop_start() def u32(inv, r): return (inv.read_input_registers(r,1)[0] << 16) | inv.read_input_registers(r+1,1)[0] def s16(v): return v if v < 32768 else v - 65536 def poll(): try: inv = PySolarmanV5(INVERTER_IP, LOGGER_SERIAL, port=8899, mb_slave_id=1) d = { "ts": datetime.now().isoformat(), "status": inv.read_input_registers(0x0000,1)[0], "fault": inv.read_input_registers(0x0001,1)[0], "dc_v": inv.read_input_registers(0x0002,1)[0] / 10, "dc_a": inv.read_input_registers(0x0003,1)[0] / 10, "ac_w": inv.read_input_registers(0x0008,1)[0], "ac_v": inv.read_input_registers(0x0006,1)[0] / 10, "freq": inv.read_input_registers(0x0009,1)[0] / 100, "temp": s16(inv.read_input_registers(0x000B,1)[0]) / 10, "today_kwh": inv.read_input_registers(0x000D,1)[0] / 10, "total_kwh": u32(inv, 0x000E) / 10, } d["efficiency"] = round(d["ac_w"] / RATED_W * 100, 1) d["saved_rs"] = round(d["today_kwh"] * TARIFF, 2) mq.publish("solar/telemetry", json.dumps(d), qos=1) if d["fault"]: mq.publish("solar/fault", json.dumps(d), qos=1) logging.info(f"✅ {d['ac_w']}W | {d['efficiency']}% | {d['today_kwh']}kWh") inv.disconnect() except NoSocketAvailableError: logging.warning("🌙 Night / unreachable") except Exception as e: logging.error(f"❌ {e}") while True: poll(); time.sleep(POLL_S)
